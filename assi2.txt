-- 1. Create Database
CREATE DATABASE StudentManagement;

-- 2. Use Database
USE StudentManagement;

-- 3. Create Students Table
CREATE TABLE Students (
    PRN INT PRIMARY KEY,
    FirstName VARCHAR(50) NOT NULL,
    LastName VARCHAR(50) NOT NULL,
    MiddleName VARCHAR(50),
    Year INT NOT NULL,
    Email VARCHAR(100) UNIQUE,
    Age INT NOT NULL,
    Department VARCHAR(100) NOT NULL
);

-- 4. Insert Records
INSERT INTO Students VALUES

(103, 'Alice', 'Johnson', NULL, 3, 'alice.johnson@example.com', 21, 'Mechanical'),
(104, 'Brod', 'Stud', 'B', 2, 'brod.a@example.com', 20, 'Electronics');

-- 5. Retrieve All Records
SELECT * FROM Students;

-- 6. Update Year of PRN 102
UPDATE Students
SET Year = 2
WHERE PRN = 102;

-- 7. Delete Student PRN 103
DELETE FROM Students
WHERE PRN = 103;

-- 9. Distinct Departments
SELECT DISTINCT Department FROM Students;

-- 10. Count Distinct Departments
SELECT COUNT(DISTINCT Department) FROM Students;

-- 11. Distinct Years
SELECT DISTINCT Year FROM Students;

-- 12. Distinct Ages (Computer Science)
SELECT DISTINCT Age
FROM Students
WHERE Department = 'Computer Science';

-- 13. Distinct First Names (Age < 21)
SELECT DISTINCT FirstName
FROM Students
WHERE Age < 21;

-- 14. Students Older Than 20
SELECT *
FROM Students
WHERE Age > 20;

-- 15. Students in Third Year
SELECT *
FROM Students
WHERE Year = 3;

-- 16. Names + Department (PRN > 101)
SELECT FirstName, LastName, Department
FROM Students
WHERE PRN > 101;

-- 17. Electronics Students (Name + Email)
SELECT FirstName, Email
FROM Students
WHERE Department = 'Electronics';

-- 18. CS Students Older Than 19
SELECT *
FROM Students
WHERE Department = 'Computer Science' AND Age > 19;

-- 19. Remove All Records (Keep Table)
TRUNCATE TABLE Students;

-- 20. Drop Table
DROP TABLE Students;
