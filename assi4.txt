-- 1. Create Tables

CREATE TABLE Customers (
    customer_id INT PRIMARY KEY,
    name VARCHAR(50),
    email VARCHAR(100),
    city VARCHAR(50),
    age INT
);

CREATE TABLE Orders (
    order_id INT PRIMARY KEY,
    customer_id INT,
    order_date DATE,
    total_amount DECIMAL(10,2),
    status VARCHAR(20),
    FOREIGN KEY (customer_id) REFERENCES Customers(customer_id)
);

CREATE TABLE Products (
    product_id INT PRIMARY KEY,
    name VARCHAR(50),
    category VARCHAR(50),
    price DECIMAL(10,2),
    stock INT
);

CREATE TABLE Order_Items (
    order_item_id INT PRIMARY KEY,
    order_id INT,
    product_id INT,
    quantity INT,
    subtotal DECIMAL(10,2),
    FOREIGN KEY (order_id) REFERENCES Orders(order_id),
    FOREIGN KEY (product_id) REFERENCES Products(product_id)
);

-- 2. Queries

-- 1. Total amount spent by customer ID 3
SELECT SUM(total_amount) AS total_spent
FROM Orders
WHERE customer_id = 3;

-- 2. Total revenue from all orders
SELECT SUM(total_amount) AS total_revenue
FROM Orders;

-- 3. Stock value for each product
SELECT name, (price * stock) AS stock_value
FROM Products;

-- 4. Customers above 30 and from New York
SELECT *
FROM Customers
WHERE age > 30 AND city = 'New York';

-- 5. Customers in LA OR orders > 500
SELECT DISTINCT c.*
FROM Customers c
LEFT JOIN Orders o ON c.customer_id = o.customer_id
WHERE c.city = 'Los Angeles' OR o.total_amount > 500;

-- 6. Products in stock and cost < 50
SELECT *
FROM Products
WHERE stock > 0 AND price < 50;

-- 7. Products with stock < 5
SELECT *
FROM Products
WHERE stock < 5;

-- 8. Orders not completed
SELECT *
FROM Orders
WHERE status <> 'Completed';

-- 9. Products out of stock OR Electronics
SELECT *
FROM Products
WHERE stock = 0 OR category = 'Electronics';

-- 10. Gmail users
SELECT email
FROM Customers
WHERE email LIKE '%@gmail.com';

-- 11. Product names containing 'Phone'
SELECT name
FROM Products
WHERE name LIKE '%Phone%';

-- 12. Customers with exactly 5 characters name
SELECT *
FROM Customers
WHERE name LIKE '_____';

-- 13. Products with Laptop or Tablet
SELECT *
FROM Products
WHERE name LIKE '%Laptop%' OR name LIKE '%Tablet%';

-- 14. Orders placed in 2024
SELECT *
FROM Orders
WHERE YEAR(order_date) = 2024;

-- 15. Customers in NY, LA, Chicago
SELECT *
FROM Customers
WHERE city IN ('New York', 'Los Angeles', 'Chicago');

-- 16. Products NOT in Electronics or Clothing
SELECT *
FROM Products
WHERE category NOT IN ('Electronics', 'Clothing');

-- 17. Orders by customers 1, 3, 5
SELECT *
FROM Orders
WHERE customer_id IN (1, 3, 5);

-- 18. Customers with no orders
SELECT *
FROM Customers c
LEFT JOIN Orders o ON c.customer_id = o.customer_id
WHERE o.order_id IS NULL;

-- 19. Orders in last 30 days
SELECT *
FROM Orders
WHERE order_date >= CURDATE() - INTERVAL 30 DAY;

-- 20. Orders where amount multiple of 100
SELECT *
FROM Orders
WHERE total_amount % 100 = 0;
