-- ==============================
-- 1. CREATE TABLES
-- ==============================

CREATE TABLE Customers (
    customer_id INT PRIMARY KEY,
    name VARCHAR(50),
    email VARCHAR(100),
    city VARCHAR(50)
);

CREATE TABLE Orders (
    order_id INT PRIMARY KEY,
    customer_id INT,
    order_date DATE,
    total_amount DECIMAL(10,2),
    FOREIGN KEY (customer_id) REFERENCES Customers(customer_id)
);

CREATE TABLE Products (
    product_id INT PRIMARY KEY,
    product_name VARCHAR(50),
    category VARCHAR(50),
    price DECIMAL(10,2)
);

CREATE TABLE Order_Items (
    order_item_id INT PRIMARY KEY,
    order_id INT,
    product_id INT,
    quantity INT,
    subtotal DECIMAL(10,2),
    FOREIGN KEY (order_id) REFERENCES Orders(order_id),
    FOREIGN KEY (product_id) REFERENCES Products(product_id)
);

CREATE TABLE Sellers (
    seller_id INT PRIMARY KEY,
    seller_name VARCHAR(50),
    city VARCHAR(50)
);

CREATE TABLE Product_Sellers (
    product_id INT,
    seller_id INT,
    PRIMARY KEY (product_id, seller_id),
    FOREIGN KEY (product_id) REFERENCES Products(product_id),
    FOREIGN KEY (seller_id) REFERENCES Sellers(seller_id)
);

-- 3. QUERIES
-- ==============================

-- 1. Customers with their orders (including no orders)
SELECT c.*, o.*
FROM Customers c
LEFT JOIN Orders o ON c.customer_id = o.customer_id;

-- 2. Orders with product names and quantities
SELECT o.order_id, p.product_name, oi.quantity
FROM Orders o
JOIN Order_Items oi ON o.order_id = oi.order_id
JOIN Products p ON oi.product_id = p.product_id;

-- 3. Total number of orders per customer
SELECT customer_id, COUNT(*) AS total_orders
FROM Orders
GROUP BY customer_id;

-- 4. Total number of products per category
SELECT category, COUNT(*) AS total_products
FROM Products
GROUP BY category;

-- 5. Orders in last 30 days with customer name
SELECT c.name, o.order_id, o.total_amount
FROM Customers c
JOIN Orders o ON c.customer_id = o.customer_id
WHERE o.order_date >= CURDATE() - INTERVAL 30 DAY;

-- 6. Sellers who sell 'Laptop'
SELECT DISTINCT s.seller_name
FROM Sellers s
JOIN Product_Sellers ps ON s.seller_id = ps.seller_id
JOIN Products p ON ps.product_id = p.product_id
WHERE p.product_name = 'Laptop';

-- 7. Customers who never placed an order
SELECT *
FROM Customers c
LEFT JOIN Orders o ON c.customer_id = o.customer_id
WHERE o.order_id IS NULL;

-- 8. Orders with amount > average
SELECT *
FROM Orders
WHERE total_amount > (SELECT AVG(total_amount) FROM Orders);

-- 9. Customers with at least 2 orders
SELECT customer_id
FROM Orders
GROUP BY customer_id
HAVING COUNT(*) >= 2;

-- 10. Top 3 most ordered products
SELECT product_id, SUM(quantity) AS total_quantity
FROM Order_Items
GROUP BY product_id
ORDER BY total_quantity DESC
LIMIT 3;

-- 11. Products sold by multiple sellers
SELECT product_id
FROM Product_Sellers
GROUP BY product_id
HAVING COUNT(DISTINCT seller_id) > 1;

-- 12. Sellers who never sold anything
SELECT *
FROM Sellers s
LEFT JOIN Product_Sellers ps ON s.seller_id = ps.seller_id
WHERE ps.product_id IS NULL;

-- 13. Products never ordered
SELECT *
FROM Products p
LEFT JOIN Order_Items oi ON p.product_id = oi.product_id
WHERE oi.order_id IS NULL;

-- 14. Customer with highest number of orders
SELECT customer_id, COUNT(*) AS total_orders
FROM Orders
GROUP BY customer_id
ORDER BY total_orders DESC
LIMIT 1;

-- 15. Customers with >5 different products
SELECT o.customer_id
FROM Orders o
JOIN Order_Items oi ON o.order_id = oi.order_id
GROUP BY o.customer_id
HAVING COUNT(DISTINCT oi.product_id) > 5;

-- 16. Products sold by >=2 sellers but never ordered
SELECT ps.product_id
FROM Product_Sellers ps
GROUP BY ps.product_id
HAVING COUNT(DISTINCT ps.seller_id) > 1
AND ps.product_id NOT IN (
    SELECT product_id FROM Order_Items
);

-- 17. Customer who spent most money
SELECT c.customer_id, c.name, SUM(o.total_amount) AS total_spent
FROM Customers c
JOIN Orders o ON c.customer_id = o.customer_id
GROUP BY c.customer_id, c.name
ORDER BY total_spent DESC
LIMIT 1;

-- 18. Customers who ordered OR live in seller city
SELECT customer_id FROM Orders
UNION
SELECT c.customer_id
FROM Customers c
JOIN Sellers s ON c.city = s.city;

-- 19. Products in stock OR ordered
SELECT product_id FROM Product_Sellers
UNION
SELECT product_id FROM Order_Items;

-- 20. Products ordered AND in stock
SELECT product_id FROM Order_Items
INTERSECT
SELECT product_id FROM Product_Sellers;

-- 21. Customers ordered AND live in seller city
SELECT customer_id FROM Orders
INTERSECT
SELECT c.customer_id
FROM Customers c
JOIN Sellers s ON c.city = s.city;

-- 22. Customers who ordered in every year
SELECT customer_id
FROM Orders
GROUP BY customer_id
HAVING COUNT(DISTINCT YEAR(order_date)) =
       (SELECT COUNT(DISTINCT YEAR(order_date)) FROM Orders);
